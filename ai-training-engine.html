function enviarReporteLumiere() {
  const cfg = {
    to: "equipo@tuagencia.com",
    subject: `✦ Reporte IA · ${hoy()}`,
    api: PropertiesService
      .getScriptProperties()
      .getProperty("LUMIERE_API_KEY")
  };

  // 1. Obtener reporte desde Lumière API
  const res = UrlFetchApp.fetch(
    "https://api.lumiere.ai/v1/report", {
    method: "POST",
    headers: {
      "Authorization": `Bearer ${cfg.api}`,
      "Content-Type": "application/json"
    },
    payload: JSON.stringify({ date: hoy() })
  });

  const data = JSON.parse(res.getContentText());

  // 2. Enviar por Gmail
  GmailApp.sendEmail(cfg.to, cfg.subject,
    "", {
      htmlBody: buildHTML(data),
      name: "Neuro AI"
    }
  );
}

function hoy() {
  return Utilities.formatDate(
    new Date(), "America/Bogota",
    "dd/MM/yyyy"
  );
}
